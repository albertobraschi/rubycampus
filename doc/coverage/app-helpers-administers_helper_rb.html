49   def label_for_setting(opts={})
50     setting = opts[:setting].to_s
51     example = opts[:example] || ''
52     width = opts[:width] || 'w100'
53     input_type = opts[:input_type] || 'inputselect'
54     mandatory = opts[:mandatory] ? 'mandatory' : ''
55     label = opts[:label] || opts[:setting].to_s
56     
57     case input_type.to_sym
58     when :inputselect || nil
59     (content_tag(:label, 
60                  (
61                  content_tag(:span,(content_tag(:span,_(label.humanize.titleize),:class => "title")) + (
62                  content_tag(:span, ' ' + _(example), :class => "example") unless example == nil) + (
63                  text_field_tag "settings[#{setting}]", Setting.send(setting), :class => "field"), 
64                  :class => "wrapper")), :for => "setting_#{setting}", :class => "#{width} #{input_type} #{mandatory}"))
65     when :inputselect_password
66     (content_tag(:label, 
67                 (
68                 content_tag(:span,(content_tag(:span,_(label.humanize.titleize),:class => "title")) + (
69                 content_tag(:span, ' ' + _(example), :class => "example") unless example == nil) + (
70                 password_field_tag "settings[#{setting}]", Setting.send(setting), :class => "field"), 
71                 :class => "wrapper")), :for => "setting_#{setting}", :class => "#{width} #{input_type} #{mandatory}"))             
72     when :radiocheck
73       (content_tag(:label, 
74                    (
75                    content_tag(:span,(content_tag(:span, ' ' + _(example), :class => "example") unless example == nil) + "<br/>" +(
76                    check_box_tag "settings[#{setting}]", 1, Setting.send(setting+"?") , :class => "select") + (
77                    hidden_field_tag "settings[#{setting}]", 0) + (
78                    content_tag(:span,_(label.humanize.titleize),:class => "title")), :class => "wrapper")), 
79                    :for => "setting_#{setting}", :class => "#{width} #{input_type} #{mandatory}"))    
80     else
81       "Input Type doesn't exist"
82     end
